# --- STEAM (Valve) ---
steampowered.com        # Main store, login, and account management.
steamcommunity.com      # Steam Overlay, Profiles, Trading, and Market.
steam-chat.com          # Friends list backend and real-time chat messages.
steam-static.com        # Static assets like images and CSS for the launcher.
steamcontent.com        # Critical for downloading game updates and workshop mods.
steamusercontent.com    # Cloud storage for screenshots and user-created files.

# --- EPIC GAMES ---
epicgames.com           # Main store and launcher authentication.
tracking.epicgames.com  # Required for achievement pops and syncing progress.
metrics.ol.epicgames.com # Cloud save synchronization and telemetry logs.
friends.ol.epicgames.com # Core Friends list service and "Social Overlay."
friends-public-service-prod06.ol.epicgames.com # Syncs friend requests/chats.
entitlement-public-service-prod08.ol.epicgames.com # Verifies game ownership.

# --- XBOX & MICROSOFT GAMING ---
xbox.com                # Main Xbox portal and web profile.
xboxlive.com            # Multiplayer servers, Party Chat, and Gamertags.
vortex.data.microsoft.com # Primary server for achievement "pops" and tracking.
v10.events.data.microsoft.com # Secondary telemetry for system achievements.
v20.events.data.microsoft.com # High-priority Windows Gaming event syncing.
msgamestudios.com       # Identity verification for Microsoft-owned games.

# --- EA APP (Electronic Arts) ---
ea.com                  # Main website and authentication server.
sdk.split.io            # Critical: If blocked, the Library view often disappears.
telemetry.ea.com        # Required for "Time Played" and Achievement tracking.
eacomms.ea.com          # EA social systems and in-app messaging.

# --- UBISOFT CONNECT ---
ubisoft.com             # Main account and store server.
ubi.com                 # Legacy login and platform authentication.
ubisoft-connect.ubi.com # Ubisoft Connect launcher synchronization.
static-ubiservices.ubi.com # Core challenges, rewards, and XP tracking.
social.cdn.ubisoft.com  # Downloads friend avatars and social list data.
overlay.ubisoft.com     # Enables the in-game social menu and messages.
avatars.ubisoft.com     # Handles user profile pictures across games.

# --- BLIZZARD (Battle.net) ---
battle.net              # Main launcher login and store.
blizzard.com            # Account management and website features.

oauth.battle.net        # Secure login verification for launcher/games.
blzstatic.com           # High-speed delivery for store images and icons.

rockstargames.com       # Main website and Social Club login.
